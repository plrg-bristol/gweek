\begin{small}
\begin{mathpar}
  \inferH{Var}{
    \strut
  }{
    \IsVTerm[\Gamma, x : A, \Gamma']{x}{A}
  }
  \and
  \inferH{Unit}{
    %
  }{
    \IsVTerm{\UnitV}{\UnitT}
  }
  \and
  \inferH{Zero}{
    \strut
  }{
    \IsVTerm{\Zero}{\NatT}
  }
  \and
  \inferH{Succ}{
    \IsVTerm{V}{\NatT}
  }{
    \IsVTerm{\Succ{V}}{\NatT}
  }
  \and
  \inferH{Pair}{
    \IsVTerm{V_1}{A_1} \\
    \IsVTerm{V_2}{A_2}
  }{
    \IsVTerm{\VTuple{V_1}{V_2}}{A_1 \times A_2}
  }
  \and
  \inferH{Nil}{
    \strut
  }{
    \IsVTerm{\Nil}{\ListT{A}}
  }
  \and
  \inferH{Cons}{
    \IsVTerm{V}{A} \\
    \IsVTerm{W}{\ListT{A}}
  }{
    \IsVTerm{\Cons{V}{W}}{\ListT{A}}
  }
  \and
  \inferH{Thunk}{
    \IsCTerm{M}{\CType{C}}
  }{
    \IsVTerm{\Thunk{M}}{\ThunkT{\CType{C}}}
  }
  \and
  \inferH{In}{
    \IsVTerm{V}{A_i}
  }{
    \IsVTerm{\In[i]{V}}{A_1 + A_2}
  }
  \\
  \inferH{Return}{
    \IsVTerm{V}{A}
  }{
    \IsCTerm{\Return{V}}{\ReturnerT{A}}
  }
  \and
  \inferH{Bind}{
    \IsCTerm{M}{\ReturnerT{A}} \\
    \IsCTerm[\Gamma, \DeclVar{x}{A}]{N}{\CType{C}}
  }{
    \IsCTerm{\CTo{M}{x}{N}}{\CType{C}}
  }
  \and
  \inferH{Lam}{
    \IsCTerm[\Gamma, \DeclVar{x}{A}]{M}{\CType{C}}
  }{
    \IsCTerm{\Lam{x}[A]{M}}{A \to \CType{C}}
  }
  \and
  \inferH{App}{
    \IsCTerm{M}{A \to \CType{C}} \\
    \IsVTerm{V}{A}
  }{
    \IsCTerm{\CApp{M}{V}}{\CType{C}}
  }
  \and
  \inferH{Force}{
    \IsVTerm{V}{\ThunkT{\CType{C}}}
  }{
    \IsCTerm{\Force{V}}{\CType{C}}
  }
  \and
  \inferH{Fix}{
    \IsCTerm[\Gamma, x : \ThunkT{\CType{C}}]{M}{\CType{C}}
  }{
    \IsCTerm{\Fix{x}[\ThunkT{\CType{C}}]{M}}{\CType{C}}
  }
  \\
  \inferH{Ifz}{
    \IsVTerm{V}{\NatT} \\
    \IsCTerm{M}{\CType{C}} \\
    \IsCTerm[\Gamma, n : \NatT]{N}{\CType{C}}
  }{
    \IsCTerm{\CIfz{V}{M}{n}{N}}{\CType{C}}
  }
  \and
  \inferH{Match}{
    \IsVTerm{V}{\ListT{A}} \\
    \IsCTerm{M}{\CType{C}} \\
    \IsCTerm[\Gamma, x : A, xs : \ListT{A}]{N}{\CType{C}}
  }{
    \IsCTerm{\CMatch{V}{M}{x}{xs}{N}}{\CType{C}}
  }
  \and
  \inferH{Split}{
    \IsVTerm{V}{A_1 \times A_2} \\
    \IsCTerm[\Gamma, x_1 : A_1, x_2 : A_2]{M}{\CType{C}}
  }{
    \IsCTerm{\Split{V}{x_1}{x_2}{M}}{\CType{C}}
  }
  \\
  \inferH{Fail}{
    \strut
  }{
    \IsCTerm{\Fail}{\CType{C}}
  }
  \and
  \inferH{Choice}{
    \IsCTerm{M_1}{\CType{C}} \\
    \IsCTerm{M_2}{\CType{C}}
  }{
    \IsCTerm{\Choice{M_1}{M_2}}{\CType{C}}
  }
  \and
  \inferH{Exists}{
    \IsCTerm[\Gamma, x : \sigma]{M}{\CType{C}}
  }{
    \IsCTerm{\Exists{x}[\sigma]{M}}{\CType{C}}
  }
  \and
  \inferH{Equate}{
    \IsVTerm{V}{\sigma} \\
    \IsVTerm{W}{\sigma} \\
    \IsCTerm{M}{\CType{C}}
  }{
    \IsCTerm{\Equate{V}{W}[\sigma]{M}}{\CType{C}}
  }
\end{mathpar}
\end{small}
